**LineRenderer.cs** -- Update `BuildScene` to produce marker segments when series has MarkerShape != None:

```csharp
public static LineRenderScene BuildScene(LineChartData data)
{
    ArgumentNullException.ThrowIfNull(data);

    var segments = new List<LineRenderSegment>();
    var markers = new List<MarkerRenderSegment>();

    foreach (var series in data.Series)
    {
        var hasMarkers = series.MarkerShape != MarkerShape.None;
        var markerColor = series.MarkerColor ?? series.Color;

        for (var i = 0; i < series.Points.Count - 1; i++)
        {
            var start = series.Points[i];
            var end = series.Points[i + 1];

            segments.Add(new LineRenderSegment(
                new Vector3((float)start.Horizontal, (float)start.Value, (float)start.Depth),
                new Vector3((float)end.Horizontal, (float)end.Value, (float)end.Depth),
                series.Width,
                start.Color ?? series.Color));

            if (hasMarkers)
            {
                markers.Add(new MarkerRenderSegment(
                    new Vector3((float)start.Horizontal, (float)start.Value, (float)start.Depth),
                    start.Color ?? markerColor,
                    series.MarkerSize,
                    series.MarkerShape));
            }
        }

        // Add marker for the last point (no segment starts from it)
        if (hasMarkers && series.Points.Count > 0)
        {
            var last = series.Points[^1];
            markers.Add(new MarkerRenderSegment(
                new Vector3((float)last.Horizontal, (float)last.Value, (float)last.Depth),
                last.Color ?? markerColor,
                series.MarkerSize,
                series.MarkerShape));
        }
    }

    return new LineRenderScene(data.SeriesCount, segments, markers);
}
```

Note: This also incorporates the per-point color fix from Plan 03 (`start.Color ?? series.Color`). If Plan 03 is executed first, this is already in place. If Plan 04 is executed first, the `start.Color ?? series.Color` pattern is included here. Either way, the final state is correct.

```csharp
/// <summary>
/// Configures marker rendering for the line series.
/// </summary>
/// <param name="shape">The marker shape.</param>
/// <param name="size">The marker size in pixels. Must be positive.</param>
/// <param name="color">Optional marker ARGB color. When null, uses each point's color or series color.</param>
public void SetMarker(MarkerShape shape, float size = 4f, uint? color = null)
{
    if (size <= 0f)
    {
        throw new ArgumentOutOfRangeException(nameof(size), "Marker size must be positive.");
    }

    var data = LineData ?? throw new InvalidOperationException("Line series requires line data.");
    var series = data.Series.ToArray();
    for (var i = 0; i < series.Length; i++)
    {
        var s = series[i];
        series[i] = new LineSeries(s.Points, s.Color, s.Width, s.Label, shape, size, color);
    }
    ReplaceLineData(new LineChartData(series, data.Metadata));
}
```
