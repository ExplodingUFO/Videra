   ```csharp

   namespace Videra.Core.Logging;

   /// <summary>
   /// Logger extension methods for structured logging with consistent templates
   /// </summary>
   public static class LoggerExtensions
   {
       // Define common log message templates as constants for consistency
       // These are examples -- add more as needed during the Console.WriteLine replacement

       [LoggerMessage(Level = LogLevel.Information, Message = "[{Component}] {Message}")]
       public static partial void LogComponentMessage(this ILogger logger, string component, string message);

       [LoggerMessage(Level = LogLevel.Debug, Message = "[{Component}] {Action}: {Details}")]
       public static partial void LogComponentAction(this ILogger logger, string component, string action, string details);

       [LoggerMessage(Level = LogLevel.Error, Message = "[{Component}] {Action} failed: {Error}")]
       public static partial void LogComponentError(this ILogger logger, string component, string action, string error);

       [LoggerMessage(Level = LogLevel.Warning, Message = "[{Component}] {Message}")]
       public static partial void LogComponentWarning(this ILogger logger, string component, string message);
   }
   ```
   IMPORTANT: Microsoft.Extensions.Logging's [LoggerMessage] source generator requires the `partial` keyword on methods. Ensure the project file has the necessary attributes. Alternatively, if source generators are not desired, use simple ILogger extension methods that call logger.LogInformation/logDebug directly. Choose whichever is simpler and compiles cleanly.

   Actually, [LoggerMessage] attribute requires Microsoft.Extensions.Logging generically. For simplicity and to avoid source generator complexity, use direct ILogger.Log* calls instead:
   ```csharp
   using Microsoft.Extensions.Logging;

   namespace Videra.Core.Logging;

   /// <summary>
   /// Logger extension methods for structured logging with consistent templates
   /// </summary>
   public static class LoggerExtensions
   {
       public static void LogComponentInfo(this ILogger logger, string component, string message)
       {
           logger.LogInformation("[{Component}] {Message}", component, message);
       }

       public static void LogComponentDebug(this ILogger logger, string component, string action, string details)
       {
           logger.LogDebug("[{Component}] {Action}: {Details}", component, action, details);
       }

       public static void LogComponentError(this ILogger logger, string component, string action, string error, Exception? ex = null)
       {
           logger.LogError(ex, "[{Component}] {Action} failed: {Error}", component, action, error);
       }

       public static void LogComponentWarning(this ILogger logger, string component, string message)
       {
           logger.LogWarning("[{Component}] {Message}", component, message);
       }
   }
   ```
